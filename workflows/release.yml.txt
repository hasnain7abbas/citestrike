name: Release
on:
  push:
    tags:
      - 'v*' # Runs when you push a tag like v1.0.0

jobs:
  publish:
    permissions:
      contents: write
    strategy:
      fail-fast: false
      matrix:
        include:
          - platform: 'windows-latest'
            args: ''

    runs-on: ${{ matrix.platform }}
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: lts/*
          cache: 'npm' # Change to 'yarn' or 'pnpm' if you use those

      - name: Install Rust
        uses: dtolnay/rust-toolchain@stable

      - name: Rust Cache
        uses: swatinem/rust-cache@v2
        with:
          workspaces: './src-tauri -> target'

      - name: Install frontend dependencies
        run: npm install # Change to yarn/pnpm if needed

      - name: Build and Release
        uses: tauri-apps/tauri-action@v0
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        with:
          tagName: v__VERSION__ # Auto-grabs version from tauri.conf.json
          releaseName: 'CiteStrike v__VERSION__'
          releaseBody: 'Automated build of CiteStrike.'
          releaseDraft: true
          prerelease: false
          args: ${{ matrix.args }}
